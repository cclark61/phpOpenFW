  146. // Add URL Parameters Function
  147. /**
  148. * Given a URL, add another paramter to it and return it.
  149. * @param string A URL
  150. * @param array An array in the form of [Key] => [Value] to be used for paramters.
  151. * @param bool Separator: [False] = '&' (default), [True] = '&amp;'
  152. * @param bool [True] = URL Encode Data (default), [False] = Do Not URL Encode Data
  153. * @return string New URL with update arguments/parameters
  154. */
  155. //*****************************************************************************
  156. //*****************************************************************************
  157. function add_url_params($in_url, $params, $xml_escape=false, $url_encode=true)
  158. {
  159.     $out_url = $in_url;
  160.     if (!is_array($params)) {
  161.         trigger_error('Error: [add_url_params] :: Second argument must be an array.', E_USER_WARNING );
  162.         return $out_url;
  163.     }
  164.     else if (count($params) <= 0) {
  165.         return $out_url;
  166.     }
  167.     else {
  168.         $args_started = false;
  169.         foreach ($params as $arg => $val) {
  170.             if (!$args_started && stristr($out_url, '?') === false) {
  171.                 $out_url .= '?';
  172.                 $args_started = true;
  173.             }
  174.             else {
  175.                 if ($xml_escape) { $out_url .= '&amp;'; }
  176.                 else { $out_url .= '&'; }
  177.             }
  178.             if (!$url_encode) { $out_url .= $arg . '=' . $val; }
  179.             else { $out_url .= $arg . '=' . urlencode($val); }
  180.         }
  181.     }
  182.     return $out_url;
  183. }

  252.     /**

  261.     {

  265.         if ($this->skip_render || (defined('POFW_SKIP_RENDER') && POFW_SKIP_RENDER)) {

  300.             }
  301.             $tmp->add_child($tmp2);

  305.         //===========================================================
  306.         // Theme CSS files

  309.             $tmp = new gen_element('theme_css_files');

  316.                     $tmp2->add_child($tmp3);
  317.                 }
  318.                 $tmp->add_child($tmp2);
  319.             }
  320.             $page->add_child($tmp);
  321.         }
  322.

  332.

  334.         // Theme Javascript files

  416.         
  417.         //===========================================================

  421.  
  422.         //===========================================================

  424.         //===========================================================
  425.         if (isset($this->content_xml) && is_array($this->content_xml) && $this->content_xml) {

  427.         }
  428.         
  429.         //===========================================================
  430.         // Render Page

  432.         $this->page_xml = $page;
  433.  
  434.         //===========================================================
  435.         // Determine output type

  438.         
  439.         //===========================================================

  441.         //===========================================================
  442.         ob_start();
  443.         $sxoe = (isset($_SESSION['show_xml_on_error']) && $_SESSION['show_xml_on_error'] == 1) ? (true) : (false);
  444.         xml_transform($this->page_xml, $this->xsl_template, $sxoe);
  445.         $page_content = ob_get_clean();
  446.  
  447.         //===========================================================
  448.         // Finally!! Output the results!!
  449.         //===========================================================
  450.         echo $page_content;
  451.  
  452.         //===========================================================
  453.         // Post-page Include Script (post_page.inc.php)
  454.         //===========================================================
  455.         $post_page_inc = "{$this->file_path}/{$this->mods_dir}/post_page.inc.php";
  456.         if (file_exists($post_page_inc)) { require_once($post_page_inc); }
  457.     }

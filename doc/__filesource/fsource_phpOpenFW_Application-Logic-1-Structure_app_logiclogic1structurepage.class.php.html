  192.         $this->site_title = (isset($_SESSION['site_title'])) ? ($_SESSION['site_title']) : ('');
  193.         $this->theme = (!empty($_SESSION['theme'])) ? ($_SESSION['theme']) : ('default');
  194.         $this->theme_dir = "{$this->file_path}/themes/{$this->theme}";
  195.         $this->creator = (isset($_SESSION['creator'])) ? ($_SESSION['creator']) : ('');
  196.         $this->time_zone = (isset($_SESSION['time_zone'])) ? ($_SESSION['time_zone']) : ('EST');
  197.         $this->account_name = (isset($_SESSION['account_name'])) ? ($_SESSION['account_name']) : (false);
  198.         $this->page_type = (isset($_POST['page_type'])) ? ( $_POST['page_type'] ) : ( (isset($_GET['page_type'])) ? ($_GET['page_type']) : ('page') );
  199.         $this->output_type = 'html';
  200.         $this->xsl_template = false;
  201.         $this->skip_render = false;
  202.         
  203.         //****************************************************
  204.         // Template Directory
  205.         //****************************************************
  206.         $tmp_tpl_path = "{$this->theme_dir}/templates";
  207.         if (is_dir($tmp_tpl_path)) {
  208.             $this->templates_dir = $tmp_tpl_path;            
  209.         }
  210.         else {
  211.             $this->templates_dir = $this->file_path . '/templates';
  212.         }
  213.  
  214.         //****************************************************
  215.         // Nav XML Format
  216.         //****************************************************
  217.         if (isset($_SESSION['nav_xml_format'])) {
  218.             $valid_formats = array('numeric' => 'numeric', 'rewrite' => 'rewrite', 'long_url' => 'long_url');
  219.             $this->nav_xml_format = (isset($valid_formats[$_SESSION['nav_xml_format']])) ? ($valid_formats[$_SESSION['nav_xml_format']]) : ('numeric');
  220.         }
  221.         else { $this->nav_xml_format = 'numeric'; }
  222.  
  223.         
  224.         //****************************************************
  225.         // Pre-page Include Script (pre_page.inc.php)

  234.

  236.         // Set Javascript files, CSS files, Application XML to an empty array

  239.         $this->css_files = array();
  240.         $this->theme_js_files = array();
  241.         $this->theme_css_files = array();
  242.         $this->site_xml = (isset($_SESSION['site_xml'])) ? ($_SESSION['site_xml']) : (false);
  243.         $this->app_xml = array();
  244.     }
  245.     
  246.     //************************************************************************************
  247.     /**
  248.     * Render the current page
  249.     * Compile all XML parts, and if an XSL stylesheet is present for this page type
  250.     * transform the XML into HTML/XHTML
  251.     **/
  252.     //************************************************************************************
  253.     // Render function
  254.     //************************************************************************************
  255.     protected function render()
  256.     {
  257.         //===========================================================
  258.         // Skip the Render Process?
  259.         //===========================================================
  260.         if ($this->skip_render || (defined('POFW_SKIP_RENDER') && POFW_SKIP_RENDER)) {
  261.             return false;
  262.         }
  263.  
  264.         //===========================================================
  265.         // Set Master XSL Template File
  266.         //===========================================================
  267.         if (!$this->xsl_template) {
  268.             $xslt_stylesheet = "{$this->templates_dir}/{$this->page_type}.xsl";
  269.             $this->set_xsl_template($xslt_stylesheet);
  270.         }
  271.  
  272.         //===========================================================
  273.         // Start Page
  274.         //===========================================================
  275.         $page = new gen_element('page');
  276.         $page->display_tree();
  277.  
  278.         //===========================================================
  279.         // Start Page and set basic values
  280.         //===========================================================
  281.         $page->add_child(new gen_element('html_path', $this->html_path));
  282.         $page->add_child(new gen_element('page_type', $this->page_type));
  283.         
  284.         //===========================================================
  285.         // CSS files
  286.         //===========================================================
  287.         $tmp = new gen_element('css_files');
  288.         $tmp->display_tree();
  289.         foreach ($this->css_files as $file_attrs) {
  290.             $tmp2 = new gen_element('css_file');
  291.             $tmp2->display_tree();
  292.             foreach ($file_attrs as $attr_key => $attr_val) {
  293.                 $tmp3 = new gen_element($attr_key, $attr_val);
  294.                 $tmp2->add_child($tmp3);
  295.             }
  296.             $tmp->add_child($tmp2);
  297.         }
  298.         $page->add_child($tmp);
  299.  
  300.         //===========================================================
  301.         // Theme CSS files
  302.         //===========================================================
  303.         if ($this->theme_css_files) {
  304.             $tmp = new gen_element('theme_css_files');
  305.             $tmp->display_tree();
  306.             foreach ($this->theme_css_files as $file_attrs) {
  307.                 $tmp2 = new gen_element('theme_css_file');
  308.                 $tmp2->display_tree();
  309.                 foreach ($file_attrs as $attr_key => $attr_val) {
  310.                     $tmp3 = new gen_element($attr_key, $attr_val);
  311.                     $tmp2->add_child($tmp3);
  312.                 }
  313.                 $tmp->add_child($tmp2);
  314.             }
  315.             $page->add_child($tmp);
  316.         }
  317.  
  318.         //===========================================================
  319.         // Javascript files
  320.         //===========================================================
  321.         $tmp = new gen_element('js_files');
  322.         $tmp->display_tree();
  323.         foreach ($this->js_files as $js_file) {
  324.             $tmp->add_child(new gen_element('js_file', $js_file));
  325.         }
  326.         $page->add_child($tmp);
  327.  
  328.         //===========================================================
  329.         // Theme Javascript files
  330.         //===========================================================
  331.         if ($this->theme_js_files) {
  332.             $tmp = new gen_element('theme_js_files');
  333.             $tmp->display_tree();
  334.             foreach ($this->theme_js_files as $js_file) {
  335.                 $tmp->add_child(new gen_element('theme_js_file', $js_file));
  336.             }
  337.             $page->add_child($tmp);
  338.         }
  339.  
  340.         //===========================================================
  341.         // Copyright
  342.         //===========================================================
  343.         $curr_year = date('Y');
  344.         $page->add_child(new gen_element('copyright', "{$curr_year} {$this->creator}"));
  345.  
  346.         //===========================================================
  347.         // Account Name
  348.         //===========================================================
  349.         if ($this->account_name) {
  350.             $page->add_child(new gen_element('account_name', $this->account_name));
  351.         }
  352.         
  353.         //===========================================================
  354.         // Site XML
  355.         //===========================================================
  356.         if ($this->site_xml) {
  357.             if (is_array($this->site_xml)) {
  358.                 $tmp_site_xml = array2xml('site_data', $this->site_xml);
  359.                 $page->add_child("{$tmp_site_xml}\n");
  360.             }
  361.             else {
  362.                 $page->add_child(new gen_element('site_data', $this->site_xml));
  363.             }
  364.         }
  365.         
  366.         //===========================================================
  367.         // Application XML
  368.         //===========================================================
  369.         if (count($this->app_xml) > 0) {
  370.             $tmp = new gen_element('application_data');
  371.             $tmp->display_tree();
  372.  
  373.             foreach ($this->app_xml as $xml_line) {
  374.                 if (is_array($xml_line[1])) {
  375.                     $new_xml_line = array2xml($xml_line[0], $xml_line[1]);
  376.                     $tmp->add_child("{$new_xml_line}\n");    
  377.                 }
  378.                 else {
  379.                     $tmp->add_child(new gen_element($xml_line[0], $xml_line[1]));
  380.                 }
  381.             }
  382.             $page->add_child($tmp);
  383.         }
  384.         
  385.         //===========================================================
  386.         // If Logged in
  387.         //===========================================================
  388.         if (isset($_SESSION['userid'])) {
  389.  
  390.             // User Info
  391.             $tmp = new gen_element('user');
  392.             $tmp->display_tree();
  393.             $tmp->add_child(new gen_element('userid', $_SESSION['userid']));
  394.             if (!isset($_SESSION['name'])) { $_SESSION['name'] = ''; }
  395.             $tmp->add_child(new gen_element('name', xml_escape($_SESSION['name'])));
  396.             $page->add_child($tmp);
  397.         
  398.             // Special links (logout)
  399.             $exit_phrase = 'Logout';
  400.             if ($_SESSION['auth_data_type'] == 'none') { $exit_phrase = 'Quit'; }
  401.             $tmp = new gen_element('userlinks');
  402.             $tmp->display_tree();
  403.             $tmp2 = new gen_element('link');
  404.             $tmp2->display_tree();
  405.             $tmp2->add_child(new gen_element('linkdesc', $exit_phrase));
  406.             $tmp2->add_child(new gen_element('linkhref', "{$this->html_path}/?mod=logout"));
  407.             $tmp->add_child($tmp2);
  408.  
  409.             $page->add_child($tmp);
  410.         }
  411.         
  412.         //===========================================================
  413.         // Menu
  414.         //===========================================================
  415.         if (isset($this->menu_xml)) { $page->add_child($this->menu_xml); }
  416.  
  417.         //===========================================================
  418.         // Content
  419.         //===========================================================
  420.         if (isset($this->content_xml) && is_array($this->content_xml) && $this->content_xml) { 
  421.             foreach ($this->content_xml as $node) { $page->add_child($node); }
  422.         }
  423.         
  424.         //===========================================================
  425.         // Render Page
  426.         //===========================================================
  427.         $this->page_xml = $page;
  428.  
  429.         //===========================================================
  430.         // Determine output type
  431.         //===========================================================
  432.         if ($this->output_type == 'xml') { $this->xsl_template = ''; }
  433.         
  434.         //===========================================================
  435.         // Transform the page content
  436.         //===========================================================
  437.         ob_start();
  438.         $sxoe = (isset($_SESSION['show_xml_on_error']) && $_SESSION['show_xml_on_error'] == 1) ? (true) : (false);
  439.         xml_transform($this->page_xml, $this->xsl_template, $sxoe);
  440.         $page_content = ob_get_clean();
  441.  
  442.         //===========================================================
  443.         // Finally!! Output the results!!
  444.         //===========================================================
  445.         echo $page_content;
  446.  
  447.         //===========================================================
  448.         // Post-page Include Script (post_page.inc.php)
  449.         //===========================================================
  450.         $post_page_inc = "{$this->file_path}/{$this->mods_dir}/post_page.inc.php";
  451.         if (file_exists($post_page_inc)) { require_once($post_page_inc); }
  452.     }

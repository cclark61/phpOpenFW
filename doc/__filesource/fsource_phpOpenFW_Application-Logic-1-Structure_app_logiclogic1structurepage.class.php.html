  178.     protected function initialize()

  181.         $this->file_path = $_SESSION['file_path'];
  182.         $this->frame_path = $_SESSION['frame_path'];
  183.         $this->app_logic_path = $_SESSION['app_logic_path'];
  184.         $this->html_path = $_SESSION['html_path'];
  185.         $this->mods_dir = (isset($_SESSION['modules_dir'])) ? ($_SESSION['modules_dir']) : ('modules');
  186.         $this->local_html_path = '';
  187.         $this->site_title = (isset($_SESSION['site_title'])) ? ($_SESSION['site_title']) : ('');
  188.         $this->theme = (!empty($_SESSION['theme'])) ? ($_SESSION['theme']) : ('default');
  189.         $this->theme_dir = "{$this->file_path}/themes/{$this->theme}";
  190.         $this->creator = (isset($_SESSION['creator'])) ? ($_SESSION['creator']) : ('');
  191.         $this->time_zone = (isset($_SESSION['time_zone'])) ? ($_SESSION['time_zone']) : ('EST');
  192.         $this->account_name = (isset($_SESSION['account_name'])) ? ($_SESSION['account_name']) : (false);
  193.         $this->page_type = (isset($_POST['page_type'])) ? ( $_POST['page_type'] ) : ( (isset($_GET['page_type'])) ? ($_GET['page_type']) : ('page') );
  194.         $this->output_type = 'html';
  195.         $this->xsl_template = false;
  196.         
  197.         //****************************************************
  198.         // Template Directory
  199.         //****************************************************
  200.         $tmp_tpl_path = "{$this->theme_dir}/templates";
  201.         if (is_dir($tmp_tpl_path)) {
  202.             $this->templates_dir = $tmp_tpl_path;            
  203.         }
  204.         else {
  205.             $this->templates_dir = $this->file_path . '/templates';
  206.         }
  207.  
  208.         //****************************************************
  209.         // Nav XML Format
  210.         //****************************************************
  211.         if (isset($_SESSION['nav_xml_format'])) {
  212.             $valid_formats = array('numeric' => 'numeric', 'rewrite' => 'rewrite', 'long_url' => 'long_url');
  213.             $this->nav_xml_format = (isset($valid_formats[$_SESSION['nav_xml_format']])) ? ($valid_formats[$_SESSION['nav_xml_format']]) : ('numeric');
  214.         }
  215.         else { $this->nav_xml_format = 'numeric'; }
  216.  
  217.         
  218.         //****************************************************
  219.         // Pre-page Include Script (pre_page.inc.php)
  220.         //****************************************************
  221.         $pre_page_inc = "$this->file_path/$this->mods_dir/pre_page.inc.php";

  223.

  230.         // Set Javascript files, CSS files, Application XML to an empty array
  231.         //****************************************************
  232.         $this->js_files = array();
  233.         $this->css_files = array();
  234.         $this->theme_js_files = array();
  235.         $this->theme_css_files = array();
  236.         $this->site_xml = (isset($_SESSION['site_xml'])) ? ($_SESSION['site_xml']) : (false);
  237.         $this->app_xml = array();
  238.     }
  239.     
  240.     //************************************************************************************
  241.     /**
  242.     * Render the current page
  243.     * Compile all XML parts, and if an XSL stylesheet is present for this page type
  244.     * transform the XML into HTML/XHTML
  245.     **/
  246.     //************************************************************************************
  247.     // Render function
  248.     //************************************************************************************
  249.     protected function render()
  250.     {            
  251.         //===========================================================
  252.         // Set Master XSL Template File
  253.         //===========================================================
  254.         if (!$this->xsl_template) {
  255.             $xslt_stylesheet = "{$this->templates_dir}/{$this->page_type}.xsl";
  256.             $this->set_xsl_template($xslt_stylesheet);
  257.         }
  258.  
  259.         //===========================================================
  260.         // Start Page
  261.         //===========================================================
  262.         $page = new gen_element('page');
  263.         $page->display_tree();
  264.  
  265.         //===========================================================
  266.         // Start Page and set basic values
  267.         //===========================================================
  268.         $page->add_child(new gen_element('html_path', $this->html_path));
  269.         $page->add_child(new gen_element('page_type', $this->page_type));
  270.         
  271.         //===========================================================
  272.         // CSS files
  273.         //===========================================================
  274.         $tmp = new gen_element('css_files');
  275.         $tmp->display_tree();
  276.         foreach ($this->css_files as $file_attrs) {
  277.             $tmp2 = new gen_element('css_file');
  278.             $tmp2->display_tree();
  279.             foreach ($file_attrs as $attr_key => $attr_val) {
  280.                 $tmp3 = new gen_element($attr_key, $attr_val);
  281.                 $tmp2->add_child($tmp3);
  282.             }
  283.             $tmp->add_child($tmp2);
  284.         }
  285.         $page->add_child($tmp);
  286.  
  287.         //===========================================================
  288.         // Theme CSS files
  289.         //===========================================================
  290.         if ($this->theme_css_files) {
  291.             $tmp = new gen_element('theme_css_files');
  292.             $tmp->display_tree();
  293.             foreach ($this->theme_css_files as $file_attrs) {
  294.                 $tmp2 = new gen_element('theme_css_file');
  295.                 $tmp2->display_tree();
  296.                 foreach ($file_attrs as $attr_key => $attr_val) {
  297.                     $tmp3 = new gen_element($attr_key, $attr_val);
  298.                     $tmp2->add_child($tmp3);
  299.                 }
  300.                 $tmp->add_child($tmp2);
  301.             }
  302.             $page->add_child($tmp);
  303.         }
  304.  
  305.         //===========================================================
  306.         // Javascript files
  307.         //===========================================================
  308.         $tmp = new gen_element('js_files');
  309.         $tmp->display_tree();
  310.         foreach ($this->js_files as $js_file) {
  311.             $tmp->add_child(new gen_element('js_file', $js_file));
  312.         }
  313.         $page->add_child($tmp);
  314.  
  315.         //===========================================================
  316.         // Theme Javascript files
  317.         //===========================================================
  318.         if ($this->theme_js_files) {
  319.             $tmp = new gen_element('theme_js_files');
  320.             $tmp->display_tree();
  321.             foreach ($this->theme_js_files as $js_file) {
  322.                 $tmp->add_child(new gen_element('theme_js_file', $js_file));
  323.             }
  324.             $page->add_child($tmp);
  325.         }
  326.  
  327.         //===========================================================
  328.         // Copyright
  329.         //===========================================================
  330.         $curr_year = date('Y');
  331.         $page->add_child(new gen_element('copyright', "$curr_year $this->creator"));
  332.  
  333.         //===========================================================
  334.         // Account Name
  335.         //===========================================================
  336.         if ($this->account_name) {
  337.             $page->add_child(new gen_element('account_name', $this->account_name));
  338.         }
  339.         
  340.         //===========================================================
  341.         // Site XML
  342.         //===========================================================
  343.         if ($this->site_xml) {
  344.             if (is_array($this->site_xml)) {
  345.                 $tmp_site_xml = array2xml('site_data', $this->site_xml);
  346.                 $page->add_child("$tmp_site_xml\n");
  347.             }
  348.             else {
  349.                 $page->add_child(new gen_element('site_data', $this->site_xml));
  350.             }
  351.         }
  352.         
  353.         //===========================================================
  354.         // Application XML
  355.         //===========================================================
  356.         if (count($this->app_xml) > 0) {
  357.             $tmp = new gen_element('application_data');
  358.             $tmp->display_tree();
  359.  
  360.             foreach ($this->app_xml as $xml_line) {
  361.                 if (is_array($xml_line[1])) {
  362.                     $new_xml_line = array2xml($xml_line[0], $xml_line[1]);
  363.                     $tmp->add_child("$new_xml_line\n");    
  364.                 }
  365.                 else {
  366.                     $tmp->add_child(new gen_element($xml_line[0], $xml_line[1]));
  367.                 }
  368.             }
  369.             $page->add_child($tmp);
  370.         }
  371.         
  372.         //===========================================================
  373.         // If Logged in
  374.         //===========================================================
  375.         if (isset($_SESSION['userid'])) {
  376.  
  377.             // User Info
  378.             $tmp = new gen_element('user');
  379.             $tmp->display_tree();
  380.             $tmp->add_child(new gen_element('userid', $_SESSION['userid']));
  381.             if (!isset($_SESSION['name'])) { $_SESSION['name'] = ''; }
  382.             $tmp->add_child(new gen_element('name', xml_escape($_SESSION['name'])));
  383.             $page->add_child($tmp);
  384.         
  385.             // Special links (logout)
  386.             $exit_phrase = 'Logout';
  387.             if ($_SESSION['auth_data_type'] == 'none') { $exit_phrase = 'Quit'; }
  388.             $tmp = new gen_element('userlinks');
  389.             $tmp->display_tree();
  390.             $tmp2 = new gen_element('link');
  391.             $tmp2->display_tree();
  392.             $tmp2->add_child(new gen_element('linkdesc', $exit_phrase));
  393.             $tmp2->add_child(new gen_element('linkhref', "$this->html_path/?mod=logout"));
  394.             $tmp->add_child($tmp2);
  395.  
  396.             $page->add_child($tmp);
  397.         }
  398.         
  399.         //===========================================================
  400.         // Menu
  401.         //===========================================================
  402.         if (isset($this->menu_xml)) { $page->add_child($this->menu_xml); }
  403.  
  404.         //===========================================================
  405.         // Content
  406.         //===========================================================
  407.         if (isset($this->content_xml) && is_array($this->content_xml) && $this->content_xml) { 
  408.             foreach ($this->content_xml as $node) { $page->add_child($node); }
  409.         }
  410.         
  411.         //===========================================================
  412.         // Render Page
  413.         //===========================================================
  414.         $this->page_xml = $page;
  415.  
  416.         //===========================================================
  417.         // Determine output type
  418.         //===========================================================
  419.         if ($this->output_type == 'xml') { $this->xsl_template = ''; }
  420.         
  421.         //===========================================================
  422.         // Transform the page content
  423.         //===========================================================
  424.         ob_start();
  425.         $sxoe = (isset($_SESSION['show_xml_on_error']) && $_SESSION['show_xml_on_error'] == 1) ? (true) : (false);
  426.         xml_transform($this->page_xml, $this->xsl_template, $sxoe);
  427.         $page_content = ob_get_clean();
  428.  
  429.         //===========================================================
  430.         // Finally!! Output the results!!
  431.         //===========================================================
  432.         echo $page_content;
  433.  
  434.         //===========================================================
  435.         // Post-page Include Script (post_page.inc.php)
  436.         //===========================================================
  437.         $post_page_inc = "$this->file_path/$this->mods_dir/post_page.inc.php";
  438.         if (file_exists($post_page_inc)) { include_once($post_page_inc); }
  439.     }

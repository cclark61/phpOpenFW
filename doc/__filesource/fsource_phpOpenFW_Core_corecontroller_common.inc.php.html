  50. function reg_data_source($ds_index, $ds_params)
  51. {
  52.     $known_params = array('type', 'server', 'port', 'source', 'user', 'pass', 'instance', 'conn_str', 'options', 'persistent', 'reuse_connection');
  53.     $optional_params = array('port' => '', 'user' => '', 'pass' => '', 'instance' => '', 'conn_str' => '', 'options' => '', 'persistent' => '', 'reuse_connection' => '');
  54.     settype($ds_index, 'string');
  55.     if (!is_array($ds_params)) {
  56.         trigger_error("Error: reg_data_source(): Index: '{$ds_index}', Second parameter must be an array.");
  57.         return 2;
  58.     }
  59.     else {
  60.         $param_count = count($known_params);
  61.         $new_data_source = array();
  62.         foreach ($known_params as $param_index) {
  63.             if (isset($ds_params[$param_index])) {
  64.                 $new_data_source[$param_index] = $ds_params[$param_index];
  65.                 $param_count--;
  66.             }
  67.             else if (isset($optional_params[$param_index])) { $param_count--; }
  68.         }
  69.  
  70.         if ($param_count > 0) {
  71.             trigger_error("Error: reg_data_source(): Index: '{$ds_index}', Incorrect parameter count in parameter array.");
  72.             return 4;
  73.         }
  74.         else {
  75.             $_SESSION[$ds_index] = $new_data_source;
  76.             return 0;
  77.         }
  78.     }
  79. }

  114. //************************************************************************************
  115. function set_plugin_folder($dir)
  116. {
  117.     //=================================================================
  118.     // Validate Directory
  119.     //=================================================================
  120.     if (!$dir || !is_dir($dir)) {
  121.         trigger_error('Error: set_plugin_folder(): Invalid directory passed.');
  122.         return false;        
  123.     }
  124.  
  125.     //=================================================================
  126.     // Get the MD5 Hash for indexing
  127.     //=================================================================
  128.     $pf_hash = md5($dir);
  129.  
  130.     //=================================================================
  131.     // Does App Plugin Index Exist?
  132.     //=================================================================
  133.     if (!isset($_SESSION['app_plugin_folder'])) {
  134.         $_SESSION['app_plugin_folder'] = array();
  135.     }
  136.     //=================================================================
  137.     // Check if plugin folder is already set
  138.     //=================================================================
  139.     else if (isset($_SESSION['app_plugin_folder'][$pf_hash])) {
  140.         return true;
  141.     }
  142.  
  143.     //=================================================================
  144.     // Addd New Plugin Folder
  145.     //=================================================================
  146.     $_SESSION['app_plugin_folder'][$pf_hash] = $dir;
  147.     return true;
  148. }

  194. // Load a Plugin from the "plugins" directory
  195. //************************************************************************************
  196. function load_plugin($plugin)
  197. {
  198.     //=================================================================
  199.     // Are there plugin folders defined?
  200.     //=================================================================
  201.     if (empty($_SESSION['app_plugin_folder'])) {
  202.         //trigger_error("Error: load_plugin(): No plugin folders are set!");
  203.         return false;
  204.     }
  205.  
  206.     //=================================================================
  207.     // Create a plugin hash
  208.     // Adjust plugin for Namespaces
  209.     //=================================================================
  210.     $plugin_hash = md5($plugin);
  211.     $plugin = str_replace('\\', '/', $plugin);
  212.  
  213.     //=================================================================
  214.     // Is the location of the plugin cached?
  215.     //=================================================================
  216.     if (isset($_SESSION['app_plugin_folder_cache'][$plugin_hash])) {
  217.         require_once($_SESSION['app_plugin_folder_cache'][$plugin_hash]);
  218.         return true;
  219.     }
  220.  
  221.     //=================================================================
  222.     // Attempt to locate and load the plugin
  223.     //=================================================================
  224.     foreach ($_SESSION['app_plugin_folder'] as $pf) {
  225.         $plugin_opts = array(
  226.             "{$pf}/{$plugin}.inc.php",
  227.             "{$pf}/{$plugin}.php",
  228.             "{$pf}/{$plugin}.class.php"
  229.         );
  230.         foreach ($plugin_opts as $tmp_plugin) {
  231.             if (file_exists($tmp_plugin)) {
  232.                 $_SESSION['app_plugin_folder_cache'][$plugin_hash] = $tmp_plugin;
  233.                 require_once($tmp_plugin);
  234.                 return true;
  235.             }
  236.         }
  237.     }
  238.  
  239.     //=================================================================
  240.     // Does the plugin exist as a full qualified file path?
  241.     //=================================================================
  242.     if (file_exists($plugin)) {
  243.         $_SESSION['app_plugin_folder_cache'][$plugin_hash] = $plugin;
  244.         require_once($plugin);
  245.         return true;
  246.     }
  247.  
  248.     //=================================================================
  249.     // Plugin Not Found
  250.     //=================================================================
  251.     //trigger_error("Error: load_plugin(): Plugin \"{$plugin}\" does not exist!");
  252.     return false;
  253. }

  182.     public function validate()
  183.     {
  184.         foreach ($this->validations as $key => $check) {
  185.             
  186.             //==============================================================
  187.             // Set / Reset variable values
  188.             //==============================================================
  189.             $var_val1 = null;
  190.             $var_val2 = null;
  191.             $vr = null;
  192.             
  193.             //==============================================================
  194.             // Pull variable value(s)
  195.             //==============================================================
  196.             if ($check[1] == 'custom') {
  197.                 $var_val1 = (isset($check[0])) ? ($check[0]) : (false);
  198.                 $var_val2 = (isset($check[3])) ? ($check[3]) : (false);
  199.             }
  200.             else {
  201.                 //-------------------------------------------------------
  202.                 // POST
  203.                 //-------------------------------------------------------
  204.                 if ($this->check_post) {
  205.                     $var_val1 = (isset($_POST[$check[0]]) && $_POST[$check[0]] != '') ? ($_POST[$check[0]]) : (null);
  206.                     $var_val2 = (isset($_POST[$check[3]]) && $_POST[$check[3]] != '') ? ($_POST[$check[3]]) : (null);
  207.                 }
  208.                 
  209.                 //-------------------------------------------------------
  210.                 // GET
  211.                 //-------------------------------------------------------
  212.                 if ($this->check_get && is_null($var_val1)) {
  213.                     $var_val1 = (is_null($var_val1) && isset($_GET[$check[0]]) && $_GET[$check[0]] != '') ? ($_GET[$check[0]]) : (null);
  214.                     $var_val2 = (is_null($var_val2) && isset($_GET[$check[3]]) && $_GET[$check[3]] != '') ? ($_GET[$check[3]]) : (null);
  215.                 }
  216.                 
  217.                 //-------------------------------------------------------
  218.                 // Check POST Sub Arrays
  219.                 //-------------------------------------------------------
  220.                 if (is_null($var_val1)) {
  221.                     foreach ($this->post_sub_arrays as $sub_array) {
  222.                         if (!is_null($var_val1)) { break; }
  223.                         $var_val1 = (is_null($var_val1) && isset($_POST[$sub_array][$check[0]]) && $_POST[$sub_array][$check[0]] != '') ? ($_POST[$sub_array][$check[0]]) : (null);
  224.                         $var_val2 = (is_null($var_val2) && isset($_POST[$sub_array][$check[3]]) && $_POST[$sub_array][$check[3]] != '') ? ($_POST[$sub_array][$check[3]]) : (null);
  225.                     }
  226.                 }
  227.             }
  228.             
  229.             //==============================================================
  230.             // Perform Validation
  231.             //==============================================================
  232.             switch ($check[1]) {
  233.  
  234.                 case 'is_not_empty':
  235.                     $vr = ($var_val1 != '');
  236.                     break;
  237.                     
  238.                 case 'is_empty':
  239.                     $vr = ($var_val1 == '');
  240.                     break;
  241.  
  242.                 case 'is_numeric':
  243.                     $vr = ($var_val1 !== '' && $var_val1 !== false) ? (is_numeric($var_val1)) : (true);
  244.                     break;
  245.  
  246.                 case 'is_not_numeric':
  247.                     $vr = ($var_val1 !== '' && $var_val1 !== false) ? (!is_numeric($var_val1)) : (true);
  248.                     break;
  249.  
  250.                 case 'fields_match':
  251.                     $vr = ($var_val1 == $var_val2);
  252.                     break;
  253.  
  254.                 case 'fields_not_match':
  255.                     $vr = ($var_val1 != $var_val2);
  256.                     break;
  257.  
  258.                 case 'is_date':
  259.                     $regex = '/^\d{1,2}\/\d{1,2}\/\d{4}$/';
  260.                     $vr = preg_match($regex, $var_val1);
  261.                     if (strlen($var_val1) > 10) { $vr = false; }
  262.                     break;
  263.                 
  264.                 case 'custom':
  265.                     if ($var_val1) { eval("\$vr = ($var_val1);"); }
  266.                     else if ($var_val2) { eval("\$vr = ($var_val2);"); }
  267.                     else { trigger_error('[Server Side Validation]::validate() - Invalid custom expression!'); }
  268.                     break;
  269.  
  270.                 case 'checkbox_is_checked':
  271.                     $vr = (!is_null($var_val1) && $var_val1 !== false && $var_val1 != '');
  272.                     break;
  273.  
  274.                 case 'radio_is_checked':
  275.                     $vr = (!is_null($var_val1) && $var_val1 !== false && $var_val1 != '');
  276.                     break;
  277.             }
  278.             
  279.             //==============================================================
  280.             // Result of current validation
  281.             //==============================================================
  282.             $this->check_status[$key] = $vr;
  283.             if (!$vr) {
  284.                 $this->validation_status = false;
  285.                 $this->num_failed_checks++; 
  286.                 if ($check[2] != '') {
  287.                     $this->fail_messages[$key] = $check[2];
  288.                 }
  289.             }
  290.             else if ($vr && is_null($this->validation_status)) {
  291.                 $this->validation_status = true;
  292.             }
  293.         }
  294.         
  295.         if ($this->debug_mode && !$this->validation_status) {
  296.             print "<pre>\n";
  297.             print_r($this->validations);
  298.             print_r($this->fail_messages);
  299.             print_r($_POST);
  300.             print "</pre>\n";
  301.         }
  302.  
  303.         return $this->validation_status;
  304.     }

  348.     //**************************************************************************************
  349.     public function display_fail_messages($xsl=false)
  350.     {
  351.         $xsl = trim((string)$xsl);
  352.  
  353.         //--------------------------------------------------------
  354.         // Explicitly use the "default" template that comes 
  355.         // with phpOpenFW
  356.         //--------------------------------------------------------
  357.         if (strtolower($xsl) == 'default') {
  358.             $this->xsl = $this->xsl_default;
  359.         }
  360.         //--------------------------------------------------------
  361.         // Use a specified template
  362.         //--------------------------------------------------------
  363.         else if (!empty($xsl) && file_exists($xsl)) {
  364.             $this->xsl = $xsl;
  365.         }
  366.         //--------------------------------------------------------

  370.             $this->xsl = $_SESSION['ssv_template'];
  371.         }
  372.         //--------------------------------------------------------
  373.         // Everything else uses the default template
  374.         //--------------------------------------------------------
  375.         // ..
  376.  
  377.         //--------------------------------------------------------
  378.         // Create XML
  379.         //--------------------------------------------------------
  380.         $xml = array2xml('failed_checks', $this->fail_messages());
  381.  
  382.         //--------------------------------------------------------
  383.         // Transform
  384.         //--------------------------------------------------------
  385.         xml_transform($xml, $this->xsl);
  386.     }

  111.     // Construct and output the form we have gathered info about.
  112.     //************************************************************************
  113.     public function render($buffer=false)
  114.     {
  115.         // Inset Value Buffer
  116.         ob_start();
  117.  
  118.         // Data Output Buffer
  119.         ob_start();
  120.         
  121.         // Form Label
  122.         if (isset($this->form_label)) {
  123.              print new gen_element('form_label', $this->form_label);
  124.         }
  125.         
  126.         // Headers
  127.         $headers_arr = array();
  128.         for ($x = 0; $x < $this->cols; $x++) {
  129.             if (isset($this->headers[$x])) { $headers_arr[] = $this->headers[$x]; }
  130.         }
  131.         print array2xml('headers', $headers_arr);
  132.         
  133.         // Button Cell Attributes
  134.         print array2xml('button_cell_attrs', $this->button_cell_attrs);
  135.         
  136.         // Button(s)
  137.         if (gettype($this->button) == 'array') {
  138.             $buttons = array();
  139.             foreach ($this->button as $key => $value) {
  140.                 if (gettype($key) == 'integer') { $key = 'button_' . $key; }
  141.                 $buttons[$key] = array('name' => $key, 'value' => $this->xml_escape($value));
  142.             }
  143.             print array2xml('buttons', $buttons);
  144.         }
  145.         else {
  146.             if ($this->button !== NULL) {
  147.                 print new gen_element('button', $this->xml_escape($this->button));
  148.             }
  149.         }
  150.         print new gen_element('columns', $this->cols);
  151.  
  152.         // Form content
  153.         $curr_cols = 0;
  154.         $colspan = 0;
  155.         $row_begin = false;
  156.         $row_end = false;
  157.  
  158.         // Hidden Form Elements
  159.         $hid_elems = array();
  160.         foreach ($this->hidden_elements as $he_key => $hid_element) {
  161.             ob_start();
  162.             $this->process_element($hid_element);
  163.             $hid_elems[] = (!empty($this->xsl_template)) ? ($this->xml_escape(ob_get_clean())) : (ob_get_clean());
  164.         }
  165.         print array2xml('hidden_elements', $hid_elems);
  166.         
  167.         // Visible Form Elements
  168.         ob_start();
  169.         $rows = 0;
  170.         foreach ($this->form_elements as $element) {
  171.             // Process the element (object -> render, text -> print, array -> process elements)
  172.             ob_start();
  173.             $this->process_element($element[0]);
  174.             $tmp_element = ob_get_clean();
  175.             
  176.             if ($element[2] == 'cell') {
  177.                 // Start ROW
  178.                 if ($curr_cols == 0) {
  179.                     ob_start();
  180.                     if ($rows % 2 == 1 && $this->alt_rows) { $this->set_row_attr($rows, 'class', 'alt'); }
  181.                     $row_attrs = (isset($this->row_attrs[$rows])) ? ($this->row_attrs[$rows]) : (array());
  182.                     $row_begin = true;
  183.                     $row_end = false;
  184.                     $rows++;
  185.                 }
  186.  
  187.                 // Build Form Element
  188.                 $colspan = $element[1] + 0;
  189.                 $fe_attrs = array('colspan' => $element[1]);
  190.                 foreach ($element[3] as $fe_attr_key => $fe_attr_val) { $fe_attrs[$fe_attr_key] = $fe_attr_val; }
  191.                 $fe_content = (!empty($this->xsl_template)) ? ($this->xml_escape($tmp_element)) : ($tmp_element);
  192.                 print new gen_element('form_element', $fe_content, $fe_attrs);
  193.                 
  194.                 // End ROW
  195.                 if ($curr_cols + $colspan >= $this->cols) {
  196.                     print new gen_element('row', ob_get_clean(), $row_attrs);
  197.                     $row_begin = false;
  198.                     $row_end = true;
  199.                     $curr_cols = 0;
  200.                 }
  201.                 else { $curr_cols += $colspan; }
  202.             }
  203.             else {
  204.                 print (!empty($this->xsl_template) && $element[2] != 'fieldset') ? ($this->xml_escape($tmp_element)) : ($tmp_element);
  205.             }
  206.         }
  207.         
  208.         // End ROW if not already terminated
  209.         if (!$row_end && $row_begin && count($this->form_elements) > 0) {
  210.             print new gen_element('row', ob_get_clean(), $row_attrs);
  211.         }        
  212.  
  213.         print new gen_element('elements', ob_get_clean());
  214.         print new gen_element('data', ob_get_clean());
  215.         $this->inset_val .= ob_get_clean();
  216.         return parent::render($buffer);
  217.     }

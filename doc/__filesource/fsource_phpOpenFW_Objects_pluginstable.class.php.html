  115.                 $curr_row++;

  117.             }
  118.         }

  154.     * Add a element to the table
  155.     * @param mixed New Table element
  156.     * @param integer Number of columns to span

  158.     //*************************************************************************
  159.     private function add_element($table_element, $num_cols=1, $attrs=null, $location='body', $type='data_cell')
  160.     {
  161.         // Process Element
  162.         $processed_content = $this->process_element($table_element);
  163.  
  164.         // Build Element Array
  165.         $tmp_element = array('content' => $processed_content, 'cols' => $num_cols, 'type' => $type);
  166.         
  167.         // Element Attributes
  168.         if (!is_null($attrs) && is_array($attrs)) {
  169.             $new_attrs = array();
  170.             foreach ($attrs as $key => $val) {
  171.                 if (!is_numeric($key)) { $new_attrs[$key] = $val; }
  172.             }
  173.             if (count($new_attrs) > 0) { $tmp_element['attrs'] = $new_attrs; } 
  174.         }
  175.  
  176.         // Push Element onto Array of Elements
  177.         array_push($this->table_elements[$location], $tmp_element);
  178.     }

  181.     /**

  185.     //*************************************************************************
  186.     private function process_element($element)
  187.     {
  188.         $return_content = '';
  189.         switch (gettype($element)) {                
  190.             case 'array':
  191.                 foreach ($element as $sub_element) {
  192.                     $return_content .= $this->process_element($sub_element);
  193.                 }
  194.                 break;
  195.  
  196.             case 'object':
  197.             default:
  198.                 $return_content = $element;
  199.                 break;
  200.         }
  201.  
  202.         // Return Processed Element
  203.         if (!empty($this->xsl_template)) { return $processed_content = xml_escape($return_content); }
  204.         else { return $return_content; }
  205.     }

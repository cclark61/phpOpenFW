  104.     * Load Database Sources Configuration Function

  108.     //************************************************************************
  109.     public function load_db_config($db_config, $force_config=false)
  110.     {
  111.         if ($force_config === true || !$this->db_config_set) {
  112.             if (file_exists($db_config)) {
  113.                 $data_arr = array();
  114.                 include($db_config);
  115.                 
  116.                 if (count($data_arr) > 0) {
  117.                     $key_arr2 = array_keys($data_arr);
  118.                     foreach ($key_arr2 as $key2){
  119.                         $reg_code = $this->reg_data_source($key2, $data_arr[$key2]);
  120.                         if (!$reg_code) { $_SESSION[$key2]['handle'] = 0; }
  121.                     }
  122.                     $this->db_config_set = true;
  123.                     $_SESSION['db_config_set'] = true;
  124.                 }
  125.                 else {
  126.                     trigger_error('Error: [phplitefw_controller]::load_db_config(): No data sources defined!');
  127.                     $this->db_config_set = false;
  128.                     $_SESSION['db_config_set'] = false;
  129.                 }
  130.             }
  131.             else {
  132.                 trigger_error('Error: [phplitefw_controller]::load_db_config(): Data Source Configuration file does not exist!');
  133.                 $this->db_config_set = false;
  134.                 $_SESSION['db_config_set'] = false;
  135.             }
  136.         }
  137.     }

  150.         if (file_exists($plugin_file1)) {
  151.             require_once($plugin_file1);
  152.             return true;

  154.         else if (file_exists($plugin_file2)) {
  155.             require_once($plugin_file2);

  157.         }

  163.                     $plugin_file5 = "{$pf}/{$plugin}.class.php";
  164.  
  165.                     if (file_exists($plugin_file3)) {
  166.                         require_once($plugin_file3);
  167.                         return true;
  168.                     }
  169.                     else if (file_exists($plugin_file4)) {
  170.                         require_once($plugin_file4);
  171.                         return true;
  172.                     }
  173.                     else if (file_exists($plugin_file5)) {
  174.                         require_once($plugin_file5);
  175.                         return true;
  176.                     }
  177.                 }
  178.             }
  179.         }
  180.  
  181.         if (file_exists($plugin)) {
  182.             require_once($plugin);
  183.             return true;

  185.  
  186.         trigger_error("Error: [phplitefw_controller]::load_plugin(): Plugin \"{$plugin}\" does not exist!");
  187.         return false;

  220.     {
  221.         foreach ($this->app_plugin_folder as $pf) {
  222.             if ($pf == $dir) { return true; }
  223.         }
  224.  
  225.         if (is_dir($dir)) {
  226.             $this->app_plugin_folder[] = $dir;
  227.             if ($remember) { $_SESSION['app_plugin_folder'][] = $dir; }
  228.             return true;
  229.         }
  230.         else {
  231.             trigger_error('Error: [phplitefw_controller]::set_plugin_folder() - Invalid directory passed.');
  232.             return false;
  233.         }

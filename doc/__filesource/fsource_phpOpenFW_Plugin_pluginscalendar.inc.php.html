  94.     public function render()
  95.     {
  96.         // Start Calendar Table
  97.         $tbl = new table();
  98.         $tbl->set_attribute('class', 'gen_calendar');
  99.         $tbl->set_columns(7);
  100.         if ($this->caption) { $tbl->caption($this->caption); }
  101.         $curr_cell = 0;
  102.         $curr_row = 0;
  103.         
  104.         // Header
  105.         if ($this->show_header) {
  106.             $header_attrs = array('class' => 'calendar_header');
  107.             $tbl->th('Mo.', 1, $header_attrs);
  108.             $tbl->th('Tu.', 1, $header_attrs);
  109.             $tbl->th('We.', 1, $header_attrs);
  110.             $tbl->th('Th.', 1, $header_attrs);
  111.             $tbl->th('Fr.', 1, $header_attrs);
  112.             $tbl->th('Sa.', 1, $header_attrs);
  113.             $tbl->th('Su.', 1, $header_attrs);
  114.         }
  115.         
  116.         // Empty Cells Before
  117.         for ($bx = 1; $bx <= $this->empty_cells_before; $bx++) {
  118.             $tbl->td('&nbsp;');
  119.             $curr_cell++;
  120.         }
  121.         
  122.         // Actual Date Cells
  123.         for ($x = 1; $x <= $this->last_month_day; $x++) {
  124.             $link_x = ($x < 10) ? ('0' . $x) : ($x);
  125.             if ($x == $this->day) { $tbl->td($x, 1, array('class' => 'selected')); }
  126.             else {
  127.                 if ($this->link_start) {
  128.                     $full_link = $this->link_start . urlencode("$this->month/$link_x/$this->year");
  129.                     $val = new gen_element('a', $x, array('href' => $full_link));
  130.                     $tbl->td($val);
  131.                 }
  132.                 else { $tbl->td($x); }
  133.             }
  134.             
  135.             // Increment Cell and/or Row
  136.             $curr_cell++;
  137.             if ($curr_cell == 7) {
  138.                 $curr_cell = 0;
  139.                 $curr_row++;
  140.             }
  141.         }
  142.         
  143.         // Empty Cells after
  144.         while ($curr_cell < 7) {
  145.             $tbl->td('&nbsp;');
  146.             $curr_cell++;
  147.         }
  148.     
  149.         // Render Calendar Table
  150.         //$tbl->set_alt_rows();
  151.         //$tbl->no_xsl();
  152.         $tbl->render();
  153.     }
  154. }

  63.     
  64.     switch ($start_format) {
  65.         case 'sql':

  67.             if (count($date_parts) < 3) { return false; }
  68.             list($year, $month, $day) = $date_parts;
  69.             
  70.             // Clean the Numbers
  71.             if ($clean_nums) {

  76.             
  77.             switch ($end_format) {
  78.                 case 'm/d/y':

  80.                     break;
  81.                     
  82.                 default:

  84.             }

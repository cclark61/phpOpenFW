  84.     // Render Function
  85.     //*************************************************************************
  86.     //*************************************************************************
  87.     public function render($buffer=false)
  88.     {
  89.         if ($buffer) { ob_start(); }
  90.         $el_cont = '';
  91.  
  92.         // Tabs
  93.         $tabs = '';
  94.         for ($i = 0; $i < $this->tabs; $i++) { $tabs .= "\t"; }
  95.  
  96.         //***************************************
  97.         // Child Nodes OR Inset Text
  98.         //***************************************
  99.         if ($this->child_nodes) {
  100.             // Child Nodes
  101.             foreach ($this->child_nodes as $node) {
  102.                 if (is_object($node) && get_class($node) == 'gen_element') {
  103.                      $node->set_tabs($this->tabs + 1);
  104.                 }
  105.                 $el_cont .= $node;
  106.             }
  107.         }
  108.         else if ($this->inset_val !== false && $this->inset_val !== '') {
  109.             // Inset Value
  110.             $el_cont .= ($this->style == 'tree') ? ("\t{$tabs}" . $this->inset_val . "\n") : ($this->inset_val);
  111.         }
  112.  
  113.         //***************************************
  114.         // Opening Tag        
  115.         //***************************************
  116.         $this->element_xml = "{$tabs}<{$this->element}";
  117.         if (isset($this->attributes)) {
  118.             foreach ($this->attributes as $key => $value) {
  119.                 $this->element_xml .= " {$key}=\"{$value}\"";
  120.             }
  121.         }
  122.  
  123.         //***************************************
  124.         // Build Closing Tag
  125.         //***************************************
  126.         $close_tag = "</{$this->element}>\n";
  127.         if ($this->style == 'tree') { $close_tag = $tabs . $close_tag; }
  128.  
  129.         //***************************************
  130.         // Content / Closing Tag
  131.         //***************************************
  132.         if ($el_cont !== false && $el_cont !== '') {
  133.             if ($this->style == 'tree') {
  134.                 $this->element_xml .= ">\n" . $el_cont . $close_tag;    
  135.             }
  136.             else { $this->element_xml .= '>' . $el_cont . $close_tag; }
  137.  
  138.         }
  139.         else if ($this->endtag) { $this->element_xml .= ">{$close_tag}"; }
  140.         else { $this->element_xml .=  " />\n"; }
  141.  
  142.         //***************************************
  143.         // Perform XML Transformation
  144.         //***************************************
  145.         $sxoe = (isset($_SESSION['show_xml_on_error']) && $_SESSION['show_xml_on_error'] == 1) ? (true) : (false);
  146.         xml_transform($this->element_xml, $this->xsl_template, $sxoe);
  147.         if ($buffer) { return ob_get_clean(); }
  148.     }

  173.     //*************************************************************************
  174.     /**
  175.     * Add, append, overwrite, or remove an attribute value to the element
  176.     * @param string Attribute name
  177.     * @param string Attribute value
  178.     **/    
  179.     //*************************************************************************
  180.     // Add an attribute
  181.     //*************************************************************************
  182.     public function attr($attr, $value=false, $append=true)
  183.     {
  184.         $key = trim((string)$attr);
  185.         if ($key == '.') { $key = 'class'; }
  186.         if ($key == '#') { $key = 'id'; }
  187.         $val = (string)$value;
  188.         $append = (bool)$append;
  189.         $key_exists = (isset($this->attributes[$key])) ? (true) : (false);
  190.         if ($key == '') { return false; }
  191.         if ($val == '') {
  192.             if (isset($this->attributes[$key])) {
  193.                 unset($this->attributes[$key]);
  194.             }
  195.             else { return false; }
  196.         }
  197.         else {
  198.             if ($append && $key_exists) {
  199.                 $this->attributes[$key] .= ' ' . $value;
  200.             }
  201.             else {
  202.                 $this->attributes[$key] = $value;
  203.             }
  204.         }
  205.         
  206.         return true;
  207.     }

  218.     public function attrs($attrs, $append=true)
  219.     {
  220.         if (!is_array($attrs)) { return false; }
  221.  
  222.         $count = 0;
  223.         foreach ($attrs as $attr => $value) {
  224.             if ($this->attr($attr, $value, $append)) { $count++; }
  225.         }
  226.  
  227.         return $count;
  228.     }

  317. /**
  318. * Creates a generic xml element with content
  319. * @package        phpOpenFW
  320. * @subpackage    Objects
  321. * @param string Element Name (ie. "div", "data", "p", etc.) Can be XML or XHTML
  322. * @param string Content inside of element
  323. * @param array An array, in the form of [key] => [value], of attributes
  324. */
  325. //*****************************************************************************
  326. class gen_element extends element
  327. {
  328.     public function __construct($element, $content=false, $attrs=false)
  329.     {
  330.         $this->element = $element;
  331.         $this->tabs = 0;
  332.         if ($content !== false && $content !== '') { $this->inset_val = $content; }
  333.         if (is_array($attrs)) {
  334.             foreach ($attrs as $key => $val) { $this->set_attribute($key, $val); }
  335.         }
  336.     }
  337. }

  353.     return false;

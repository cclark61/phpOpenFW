  130.             //-----------------------------------------------------------------

  139.                 return 0;
  140.             }

  142.         //=============================================
  143.         // Load data from defaults

  219.                 // Check for a default save value

  221.                 else { $this->unset_fields[$field] = ''; }

  231.     public function save($pkey_values='', $pre_args=array(), $post_args=array())
  232.     {
  233.         // Check for pre_save()
  234.         if (method_exists($this, 'pre_save')) {
  235.             if (!is_array($pre_args)) {
  236.                 $pre_args = array($pre_args);                
  237.             }
  238.             call_user_func_array(array($this, 'pre_save'), $pre_args);
  239.         }
  240.     
  241.         $qa = array();
  242.         $ret_val = false;
  243.         
  244.         // Set Table
  245.         if (isset($this->schema)) { $qa['table'] = "$this->schema.$this->table"; }
  246.         else { $qa['table'] = $this->table; }
  247.         
  248.         foreach ($this->data as $field => &$value) {
  249.             // Set to Save by default
  250.             $save = true;
  251.             
  252.             // Check if field is not supposed to save
  253.             if ($this->table_info[$field]['no_save']) {
  254.                 $save = false;
  255.             }
  256.             // Else if value is empty (but exists)
  257.             else if ($value == '') {
  258.                 // If Save Default set
  259.                 if (isset($this->table_info[$field]['save_default'])) {
  260.                     $value = $this->table_info[$field]['save_default'];
  261.                 }
  262.                 // Else if field data type save default set
  263.                 else if (array_key_exists($this->table_info[$field]['data_type'], $this->save_default_types)) {
  264.                     $value = $this->save_default_types[$this->table_info[$field]['data_type']];
  265.                 }
  266.                 // Else if field set to no save on empty
  267.                 else if (isset($this->table_info[$field]['no_save_empty']) && $this->table_info[$field]['no_save_empty']) {
  268.                     $save = false;
  269.                 }
  270.                 // Else if field data type set to no save on empty
  271.                 else if (array_key_exists($this->table_info[$field]['data_type'], $this->no_save_empty_types)) {
  272.                     $save = false;
  273.                 }
  274.                 
  275.                 // If NULL
  276.                 if (is_null($value)) {
  277.                     $value = 'NULL';
  278.                     $this->table_info[$field]['quotes'] = 'disable';
  279.                 }
  280.             }
  281.             
  282.             // Check if field is not supposed to save
  283.             if ($save) {
  284.                 // Use Bind Parameters
  285.                 if ($this->use_bind_params && $this->table_info[$field]['can_bind_param']) {
  286.     
  287.                     switch ($this->db_type) {
  288.     
  289.                         case 'mysqli':
  290.                             $qa['fields'][$field] = '?';
  291.                             $this->bind_param_count++;
  292.                             $this->bind_params[] = &$value;
  293.                             $tmp_type = (isset($GLOBALS['mysql_bind_types'][strtoupper($this->table_info[$field]['data_type'])])) 
  294.                                 ? ($GLOBALS['mysql_bind_types'][strtoupper($this->table_info[$field]['data_type'])]) 
  295.                                 : ('s');
  296.                             $this->bind_params[0] .= $tmp_type;
  297.                             break;
  298.  
  299.                         case 'sqlsrv':
  300.                             $this->bind_param_count++;
  301.                             $qa['fields'][$field] = '?';
  302.                             $this->bind_params[] = &$value;
  303.                             break;
  304.  
  305.                         case 'pgsql':
  306.                             $this->bind_param_count++;
  307.                             $tmp_param = '$' . $this->bind_param_count;
  308.                             $qa['fields'][$field] = $tmp_param;
  309.                             $this->bind_params[] = $value;
  310.                             break;
  311.         
  312.                         case 'oracle':
  313.                             $this->bind_param_count++;
  314.                             $tmp_param = 'p' . $this->bind_param_count;
  315.                             $qa['fields'][$field] = ':' . $tmp_param;
  316.                             $this->bind_params[$tmp_param] = $value;
  317.                             break;
  318.         
  319.                         case 'mssql':
  320.                         case 'mysql':
  321.                         case 'sqlite':
  322.                             switch ($this->table_info[$field]['quotes']) {
  323.                                 // Force quotes
  324.                                 case 'force':
  325.                                     $qa['fields'][$field] = "'$value'";
  326.                                     break;
  327.                                 
  328.                                 // Disable quotes
  329.                                 case 'disable':
  330.                                     $qa['fields'][$field] = $value;
  331.                                     break;
  332.                                     
  333.                                 // Auto detect if quotes are needed
  334.                                 default:
  335.                                     if (isset($this->quoted_types[$this->db_type][$this->table_info[$field]['data_type']])) {
  336.                                         $qa['fields'][$field] = "'$value'";
  337.                                     }
  338.                                     else {
  339.                                         $qa['fields'][$field] = $value;
  340.                                     }
  341.                                     break;
  342.                             }
  343.                             break;
  344.         
  345.                         case 'db2':
  346.                             $qa['fields'][$field] = '?';
  347.                             $this->bind_params[] = $value;
  348.                             $this->bind_param_count++;
  349.                             break;
  350.     
  351.                         default:
  352.         
  353.                     }
  354.                     
  355.                 }
  356.                 // Do NOT use Bind Parameters
  357.                 else {
  358.                     switch ($this->table_info[$field]['quotes']) {
  359.                         // Force quotes
  360.                         case 'force':
  361.                             $qa['fields'][$field] = "'$value'";
  362.                             break;
  363.                         
  364.                         // Disable quotes
  365.                         case 'disable':
  366.                             $qa['fields'][$field] = $value;
  367.                             break;
  368.                             
  369.                         // Auto detect if quotes are needed
  370.                         default:
  371.                             if (isset($this->quoted_types[$this->db_type][$this->table_info[$field]['data_type']])) {
  372.                                 $qa['fields'][$field] = "'$value'";
  373.                             }
  374.                             else {
  375.                                 $qa['fields'][$field] = $value;
  376.                             }
  377.                             break;
  378.                     }
  379.                 }
  380.             }
  381.         }
  382.         
  383.         // Set Query Type (insert or update)
  384.         if (!empty($pkey_values)) {
  385.             $qa['type'] = 'update';
  386.             $qa['filter_phrase'] = $this->build_where($pkey_values);
  387.         }
  388.         else {
  389.             $qa['type'] = 'insert';
  390.         }
  391.         
  392.         $query = new data_query($qa);
  393.         $strsql = $query->render();
  394.  
  395.         if ($this->print_trans) {
  396.             $ret_val = $strsql;
  397.             if ($this->use_bind_params) {
  398.                 ob_start();
  399.                 print "<br/><pre>\n";
  400.                 print_r($this->bind_params);
  401.                 print "</pre>\n";
  402.                 $ret_val .= ob_get_clean();
  403.             }
  404.         }
  405.         else {
  406.             // Create a new data transaction and execute query
  407.             $data1 = new data_trans($this->data_source);
  408.  
  409.             // Use Bind Parameters
  410.             if ($this->use_bind_params) {
  411.                 // Prepare Query
  412.                 $prep_status = $data1->prepare($strsql);
  413.  
  414.                 // Execute Query
  415.                 $exec_status = $data1->execute($this->bind_params);
  416.                 $ret_val = $exec_status;
  417.  
  418.                 // Reset Bind Variables
  419.                 $this->reset_bind_vars();                
  420.             }
  421.             // Do NOT Use Bind Parameters
  422.             else {
  423.                 $query_result = $data1->data_query($strsql);
  424.                 $ret_val = $query_result;
  425.             }
  426.  
  427.             // Lsat Insert ID if Insert Statement performed and a valid ID is returned
  428.             if ($qa['type'] == 'insert') {
  429.                 $lii = $data1->last_insert_id();
  430.                 if ($lii !== false) { $ret_val = $lii; }
  431.             }
  432.         }
  433.  
  434.         // Check for post_save()
  435.         if (method_exists($this, 'post_save')) {
  436.             if (!is_array($post_args)) {
  437.                 $post_args = array($post_args);                
  438.             }
  439.             call_user_func_array(array($this, 'post_save'), $post_args);
  440.         }
  441.  
  442.         return $ret_val;
  443.     }

  453.                 $pre_args = array($pre_args);                
  454.             }

  456.         }
  457.         
  458.         if (!empty($pkey_values)) {
  459.             $qa = array();
  460.             $qa['type'] = 'delete';
  461.  
  462.             // Set Table
  463.             if (isset($this->schema)) { $qa['table'] = "$this->schema.$this->table"; }
  464.             else { $qa['table'] = $this->table; }
  465.             $qa['filter_phrase'] = $this->build_where($pkey_values);
  466.         
  467.             $query = new data_query($qa);

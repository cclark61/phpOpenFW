  237.     public function render()

  240.         //-------------------------------------------------------------

  246.  
  247.         //-------------------------------------------------------------

  256.  
  257.         //-------------------------------------------------------------
  258.         // Escape Data (or not)
  259.         //-------------------------------------------------------------
  260.         if ($this->data_format == 'xml') {
  261.             foreach ($this->data as $dkey => &$dval) {
  262.                 if (!isset($this->no_escape_elements[$dkey])) {
  263.                     $dval = xml_escape_array($dval);
  264.                 }
  265.             }
  266.         }
  267.  
  268.         //-------------------------------------------------------------
  269.         // Create Data
  270.         //-------------------------------------------------------------
  271.         if ($this->data_format == 'xml') {
  272.             $data = array2xml($this->root_node, $this->data);
  273.         }
  274.         else if ($this->data_format == 'json') {
  275.             $data = json_encode($this->data);
  276.         }
  277.         else {
  278.             $data = $this->data;
  279.         }
  280.  
  281.         //-------------------------------------------------------------
  282.         // Output
  283.         //-------------------------------------------------------------
  284.         if ($this->show_data_only) {
  285.             if (is_array($data)) {
  286.                 print_array($data);
  287.             }
  288.             else {
  289.                 print $data;
  290.             }
  291.             return true;
  292.         }
  293.         else {
  294.             $render_function = $this->get_data('render-function');
  295.  
  296.             //----------------------------------------------------
  297.             // XML
  298.             //----------------------------------------------------
  299.             if ($this->data_format == 'xml') {
  300.                 if ($render_function) {
  301.                     return $render_function($data, $this->template);
  302.                 }
  303.                 else {
  304.                     if (file_exists($this->template)) {
  305.                         return xml_transform($data, $this->template);
  306.                     }
  307.                     else {
  308.                         if (empty($this->template)) {
  309.                             print "No template file has been specified.";
  310.                         }
  311.                         else {
  312.                             print "Invalid template file specified.";
  313.                         }
  314.                     }
  315.                 }
  316.             }
  317.             //----------------------------------------------------
  318.             // JSON / Array
  319.             //----------------------------------------------------
  320.             else {
  321.                 if ($render_function) {
  322.                     return $render_function($data, $this->template);
  323.                 }
  324.                 else {
  325.                     print "No valid render function was specified.";
  326.                 }                
  327.             }
  328.         }
  329.  
  330.         return false;
  331.     }

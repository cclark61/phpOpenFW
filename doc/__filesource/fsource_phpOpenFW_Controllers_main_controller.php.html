  257. function get_html_path()
  258. {
  259.     $path = '';
  260.     if (isset($_SERVER['DOCUMENT_ROOT']) && isset($_SERVER['SCRIPT_FILENAME'])) {
  261.         $doc_root = $_SERVER['DOCUMENT_ROOT'];
  262.         $doc_root_parts = explode('/', $doc_root);
  263.         $script_file = $_SERVER['SCRIPT_FILENAME'];
  264.         $script_file_parts = explode('/', $script_file);
  265.  
  266.         foreach ($script_file_parts as $key => $part) {
  267.             if (!isset($doc_root_parts[$key])) {
  268.                 if ($part != 'index.php') { $path .= '/' . $part; }
  269.             }
  270.         }
  271.     }
  272.     else {
  273.         $_SESSION['html_path'] = $path;
  274.         $self = $_SERVER['PHP_SELF'];
  275.         $self_arr = explode('/', $self);
  276.         foreach ($self_arr as $item) {
  277.             if (!empty($item) && $item != 'index.php') { $path .= "/$item"; }
  278.         }
  279.         if ($path == '/') { $path = ''; }
  280.     }
  281.     return $path;
  282. }

  300.             return true;
  301.         }
  302.         else if (file_exists($plugin_file4)) {

  305.         }
  306.         else if (file_exists($plugin_file5)) {

  309.         }    
  310.     }

  312.     if (file_exists($plugin)) {

  316.

  19.  * Transform an XML string given an XSL template

  21.  * @param string File path to an XSL template
  22.  */

  24. function xml_transform($xml_data, $xsl_template, $show_xml_on_error=false, $use_cache=true)
  25. {
  26.     // Load status variables
  27.     $xml_load_status = '';
  28.     $xsl_load_status = '';
  29.  
  30.     // Transform XML
  31.     if (isset($xml_data) && isset($xsl_template) && !empty($xml_data) && !empty($xsl_template)) {
  32.         
  33.         // Load the XML source
  34.         $xml = new DOMDocument();
  35.         $xml_load_status = $xml->loadXML($xml_data);
  36.         if (!$xml_load_status) {
  37.             trigger_error('Error: Malformed XML data given!!'); 
  38.             if ($show_xml_on_error) { echo $xml_data; }
  39.         }
  40.  
  41.         // Load the XSL Source
  42.         if (file_exists($xsl_template)) {
  43.             set_error_handler('HandleXSLError');
  44.             $xsl = new DOMDocument;
  45.             $xsl_load_status = $xsl->load($xsl_template, LIBXML_NOCDATA);
  46.             restore_error_handler();
  47.         }
  48.         else {
  49.             trigger_error("Error: xml_transform(): XSL Stylesheet '{$xsl_template}' does not exist!");
  50.             return false;
  51.         }
  52.  
  53.         // XSLT Processor (or XSLTCache) Object
  54.         if (extension_loaded('xslcache') && $use_cache) {
  55.             if ($xsl_load_status && $xml_load_status) { $proc = new xsltCache; }
  56.             else { $proc = new XSLTProcessor; }
  57.         }
  58.         else { $proc = new XSLTProcessor; }
  59.         $proc->registerPHPFunctions();
  60.  
  61.         // Set the XSL Stylesheet and configure processor parameters
  62.         if ($xsl_load_status) {
  63.             if (extension_loaded('xslcache') && $use_cache) { $proc->importStyleSheet($xsl_template); }
  64.             else { $proc->importStyleSheet($xsl); }
  65.         }
  66.         else {
  67.             trigger_error('Error: xml_transform(): XSL Stylesheet syntax errors occurred on load!!');
  68.             return false;
  69.         }
  70.         
  71.         // Transform XML
  72.         if ($xml_load_status && $xsl_load_status) {
  73.             $output = $proc->transformToXML($xml);
  74.  
  75.             // Check for successful output
  76.             if ($output) { echo $output; }
  77.             else {
  78.                 trigger_error('Error: xml_transform(): XML Transformation Error!!');
  79.                 echo "<br/>\n{$xml_data}";
  80.                 return false;
  81.             }
  82.         }
  83.     }
  84.     // Return Data Only
  85.     else if (isset($xml_data) && !empty($xml_data)) { echo $xml_data; }
  86.     
  87.     return true;
  88. }

  107. /**
  108.  * Generate XML Data from an array
  109.  *
  110.  * @param string Top level XML data element name
  111.  * @param array Array to generate XML from
  112.  * @param string Prefix to use for numeric element names
  113.  */
  114. //*****************************************************************************
  115. function array2xml($element, $data, $num_prefix='data_', $depth=0, $count=0)
  116. {
  117.     // Prefix numeric elements
  118.     if (is_numeric($element)) { $element = $num_prefix . $depth . '_' . $count; }
  119.     
  120.     // Create Indent Tabs
  121.     $indent = '';
  122.     for ($x = 0; $x < $depth; $x++) {
  123.         $indent .= "\t";
  124.     }
  125.  
  126.     if ($element === '') {
  127.         trigger_error("Error: generate_xml(): XML Element name not supplied! (depth: {$depth})");
  128.         return false;
  129.     }
  130.     
  131.     if (is_array($data)) {
  132.         $xml = "{$indent}<{$element}>\n";
  133.         $count = 0;
  134.         foreach ($data as $key => $row_data) {
  135.             $xml .= array2xml($key, $row_data, $num_prefix, $depth+1, $count);
  136.             $count++;    
  137.         }
  138.         $xml .= "{$indent}</{$element}>\n";    
  139.     }
  140.     else {
  141.         $xml = "{$indent}<{$element}>{$data}</{$element}>\n";
  142.     }
  143.  
  144.     return $xml;
  145. }

  148. /**

  155. function xml_escape($str_data)
  156. {
  157.     if ($str_data !== '') {
  158.         return '<![CDATA[' . strip_cdata_tags($str_data) . ']]>';
  159.     }
  160.     else { return false; }
  161. }

  164. /**
  165.  * Return an array with valid XML escaped values
  166.  *
  167.  * @param array Data to escape
  168.  * @return array Escaped data

  170. //*****************************************************************************
  171. function xml_escape_array($in_data)
  172. {
  173.     if (is_array($in_data)) {
  174.         foreach ($in_data as $key => $item) {
  175.             $in_data[$key] = xml_escape_array($item);
  176.         }
  177.         return $in_data;
  178.     }
  179.     else if ($in_data !== '') {
  180.         return '<![CDATA[' . strip_cdata_tags($in_data) . ']]>';
  181.     }
  182.     else { return false; }
  183. }

  186. /**
  187.  * Strip all CDATA begin and end tags from the string passed.
  188.  *
  189.  * @param string String to strip CDATA tags from
  190.  * @return string Cleaned string
  191.  */
  192. //*****************************************************************************
  193. function strip_cdata_tags($str_data)
  194. {
  195.     settype($str_data, 'string');
  196.     $str_data = str_replace('<![CDATA[', '', $str_data);
  197.     $str_data = str_replace(']]>', '', $str_data);
  198.     return $str_data;
  199. }

  422.     //****************************************************************************
  423.     public function render($buffer=false)
  424.     {
  425.         //---------------------------------------------------
  426.         // Buffer?
  427.         //---------------------------------------------------
  428.         if ((bool)$buffer) { ob_start(); }
  429.  
  430.         //---------------------------------------------------
  431.         // Data Order
  432.         //---------------------------------------------------
  433.         $order = array();
  434.         foreach ($this->data_outline as $key => $val) {
  435.             array_push($order, $key);
  436.         }
  437.  
  438.         //---------------------------------------------------
  439.         // Start Table Buffering if NOT Grouping
  440.         //---------------------------------------------------
  441.         ob_start();
  442.         if (!$this->group_by) {
  443.             $table_label = $this->label;
  444.             ob_start();
  445.         }
  446.  
  447.         //---------------------------------------------------
  448.         // Loop Through Rows
  449.         //---------------------------------------------------
  450.         $row_count = 0;
  451.         $gbv = NULL;
  452.         foreach ($this->record_set as $row_key => $row) {
  453.  
  454.             //---------------------------------------------------
  455.             // Set row_key to string
  456.             //---------------------------------------------------
  457.             settype($row_key, 'string');
  458.  
  459.             //---------------------------------------------------
  460.             // Start Group By Table (if group_by being used)
  461.             //---------------------------------------------------
  462.             if ($this->group_by && isset($row[$this->group_by]) && $gbv !== $row[$this->group_by]) {
  463.  
  464.                 //---------------------------------------------------
  465.                 // End Current Table
  466.                 //---------------------------------------------------
  467.                 if (!is_null($gbv)) {
  468.                     print $this->table(ob_get_clean(), $table_label, $row_count);
  469.                 }
  470.  
  471.                 //---------------------------------------------------
  472.                 // Start New Table
  473.                 //---------------------------------------------------
  474.                 ob_start();
  475.                 $table_label = isset($this->group_by_vals[$row[$this->group_by]]) ? ($this->group_by_vals[$row[$this->group_by]]) : ($row[$this->group_by]);
  476.                 $row_count = 0;
  477.                 $gbv = $row[$this->group_by];
  478.             }
  479.  
  480.             //---------------------------------------------------
  481.             // Start Row Content
  482.             //---------------------------------------------------
  483.             ob_start();
  484.  
  485.             //---------------------------------------------------
  486.             // Loop through Row Cells
  487.             //---------------------------------------------------
  488.             $cell_count = 0;
  489.             foreach ($order as $key => $val) {
  490.  
  491.                 //---------------------------------------------------
  492.                 // Print Cell
  493.                 //---------------------------------------------------
  494.                 print $this->body_cell($row_key, $val, $row[$val]);
  495.                 $cell_count++;
  496.             }
  497.  
  498.             //---------------------------------------------------
  499.             // Print Row
  500.             //---------------------------------------------------
  501.             print $this->body_row($row_key, $row_count, ob_get_clean());
  502.             $row_count++;
  503.         }
  504.         
  505.         //---------------------------------------------------
  506.         // End Table
  507.         //---------------------------------------------------
  508.         print $this->table(ob_get_clean(), $table_label, $row_count);
  509.         $this->xml = ob_get_clean();
  510.  
  511.         //---------------------------------------------------

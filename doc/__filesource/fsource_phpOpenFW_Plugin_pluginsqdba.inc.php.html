  269. function qdb_lookup($data_source, $sql, $fields='', $bind_params=false, $opts=false)
  270. {
  271.     // Check if fields are not specified
  272.     if ($fields == '') {
  273.         trigger_error('ERROR: qdb_lookup(): No return fields specified!!');
  274.     }
  275.  
  276.     // New Data Transaction
  277.     $data1 = new data_trans($data_source);
  278.     if (!empty($opts['debug'])) { $data1->data_debug(true); }
  279.     $data1->set_opt('make_bind_params_refs', 1);
  280.  
  281.     // Use Bind Parameters
  282.     if (is_array($bind_params) && count($bind_params)) {
  283.     
  284.         // Prepare Query
  285.         $prep_status = $data1->prepare($sql);
  286.     
  287.         // Execute Query
  288.         $exec_status = $data1->execute($bind_params);
  289.     }
  290.     // Straight Query
  291.     else {
  292.         // Execute Query
  293.         $query_result = $data1->data_query($sql);
  294.     }
  295.  
  296.     // Pull result set
  297.     $result = $data1->data_assoc_result();
  298.     
  299.     // If result set empty, return false
  300.     if (count($result) <= 0) {
  301.         return false;
  302.     }
  303.     // Else Check field values
  304.     else {
  305.         // Multiple fields specified
  306.         if (is_array($fields)) {
  307.             $return_vals = array();
  308.             foreach ($fields as $index) {
  309.                 if (array_key_exists($index, $result[0])) {
  310.                     $return_vals[$index] = $result[0][$index];
  311.                 }
  312.                 else {
  313.                     trigger_error("ERROR: qdb_lookup(): Field '$index' does not exist in record set!!");
  314.                 }
  315.             }
  316.         }
  317.         // One field only
  318.         else {
  319.             if (array_key_exists($fields, $result[0])) { return $result[0][$fields]; }
  320.             else {
  321.                 trigger_error("ERROR: qdb_lookup(): Field '$fields' does not exist in record set!!");
  322.             }
  323.         }
  324.     }
  325. }

  328. /**
  329. * Recordset Trim Function
  330. * 
  331. * @param array The recordset to trim
  332. * @param bool True = Trim the data (default), False = Do not trim data
  333. * @param bool True = Trim the record row keys, False = Do not trim record row keys (default)
  334. * @param bool True = Trim the record column keys, False = Do not trim record column keys (default)
  335. * @return Array Modified Record Set
  336. */
  337. //**************************************************************************************//
  338. function rs_trim($rs, $trim_data=true, $trim_row_keys=false, $trim_col_keys=false)
  339. {
  340.     if (!is_array($rs)) {
  341.         trigger_error("ERROR: rs_data_trim(): Recordset must be an array!");
  342.         return false;
  343.     }
  344.  
  345.     foreach ($rs as $key => $val) {
  346.         if (is_array($val)) {
  347.             foreach ($val as $key2 => $val2) {
  348.                 // Trim Column Keys
  349.                 if ($trim_col_keys) {
  350.                     unset($val[$key2]);
  351.                     // Trim Data
  352.                     if ($trim_data) { $val[trim($key2)] = trim($val2); }
  353.                     // Do Not Trim Data
  354.                     else { $val[trim($key2)] = $val2; }
  355.                 }
  356.                 // Do not Trim Column Keys
  357.                 // Trim Data
  358.                 else if ($trim_data) { $val[$key2] = trim($val2); }
  359.             }
  360.         }
  361.         // Trim Data
  362.         else if ($trim_data) { $val = trim($val); }
  363.  
  364.         // Trim Row Keys
  365.         if ($trim_row_keys) {
  366.             unset($rs[$key]);
  367.             $rs[trim($key)] = $val;
  368.         }
  369.         // Do not Trim Row Keys
  370.         // If Trim Data or Trim Column Keys, reset in recordset array
  371.         else if ($trim_col_keys || $trim_data) { $rs[$key] = $val; }
  372.     }
  373.  
  374.     return $rs;
  375. }

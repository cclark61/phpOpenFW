  93.         // Form Label
  94.         //============================================================

  99.         //============================================================

  104.  
  105.         foreach ($this->hidden_elements as $hid_element) {

  115.  
  116.         //============================================================

  119.         $elements = new gen_element('form_elements');

  121.  
  122.         foreach ($this->form_elements as $element) {
  123.  
  124.             //-----------------------------------------------------
  125.             // Validate Each Element
  126.             //-----------------------------------------------------
  127.             if (!is_array($element) || (is_array($element) && count($element) < 2)) { continue; }
  128.  
  129.             //-----------------------------------------------------
  130.             // Process the element 
  131.             //-----------------------------------------------------
  132.             ob_start();
  133.             $this->process_element($element[0]);
  134.             $tmp_element = ob_get_clean();
  135.  
  136.             //-----------------------------------------------------
  137.             // Build Form Element into XML and add it to Tree
  138.             //-----------------------------------------------------
  139.             $fe_attrs = array();
  140.             $fe_content = trim($tmp_element);
  141.             $content_empty = ($fe_content == '') ? (true) : (false);
  142.             if ($fe_content != '' && !empty($this->xsl_template)) {
  143.                 $fe_content = xml_escape($fe_content);
  144.             }
  145.             $fe_attrs = (isset($element[2]) && is_array($element[2])) ? ($element[2]) : (array());
  146.             $tmp_fe = new gen_element($element[1], $fe_content, $fe_attrs);
  147.             if (substr($element[1], 0, 4) != 'end_' && !$content_empty) {
  148.                 $tmp_fe->display_tree();
  149.             }
  150.             $elements->add_child($tmp_fe);
  151.         }
  152.         $this->add_child($elements);
  153.  
  154.         //============================================================
  155.         // Call Parent Render Function
  156.         //============================================================
  157.         return parent::render($buffer);
  158.     }

  168.     protected function process_element($element)
  169.     {
  170.         switch (gettype($element)) {
  171.             case 'object':
  172.                 if (method_exists($element, 'render')) {
  173.                     $element->render();
  174.                 }
  175.                 break;
  176.  
  177.             case 'array':
  178.                 foreach ($element as $sub_element) {
  179.                     $this->process_element($sub_element);
  180.                 }
  181.                 break;
  182.  
  183.             default:
  184.                 print "{$element}\n";
  185.                 break;
  186.         }

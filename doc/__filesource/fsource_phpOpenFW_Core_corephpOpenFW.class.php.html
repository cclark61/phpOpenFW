  101.                 return $_SERVER['REQUEST_URI'];

  103.             else {
  104.                 return substr($_SERVER['REQUEST_URI'], 0, $qs_start);

  107.     
  108.         return false;

  119.     {
  120.         $path = '';
  121.         if (isset($_SERVER['DOCUMENT_ROOT']) && isset($_SERVER['SCRIPT_FILENAME'])) {
  122.             $doc_root = $_SERVER['DOCUMENT_ROOT'];
  123.             $doc_root_parts = explode('/', $doc_root);
  124.             $script_file = $_SERVER['SCRIPT_FILENAME'];
  125.             $script_file_parts = explode('/', $script_file);
  126.     
  127.             foreach ($script_file_parts as $key => $part) {
  128.                 if (!isset($doc_root_parts[$key])) {
  129.                     if ($part != 'index.php') { $path .= '/' . $part; }
  130.                 }
  131.             }
  132.         }
  133.         else {
  134.             $_SESSION['html_path'] = $path;
  135.             $self = $_SERVER['PHP_SELF'];
  136.             $self_arr = explode('/', $self);
  137.             foreach ($self_arr as $item) {
  138.                 if (!empty($item) && $item != 'index.php') { $path .= "/$item"; }
  139.             }
  140.             if ($path == '/') { $path = ''; }
  141.         }
  142.         return $path;
  143.     }
